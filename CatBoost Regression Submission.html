AirBnB Optimal Market Price Regressor

After building and tuning over 300 models, this catboost regressor was the most accurate with the lowest RMSE. It is trained on a dataset featuring raw AirBnB listing data. As you can see below, the data required quite a bit of cleaning and imputation before I could go ahead and train the model.

#The code that reads the data and implements the preprocessing steps (selecting columns, cleaning, grouping etc.) here
train = pd.read_csv('train_regression 3.csv')
test = pd.read_csv('test_regression.csv')

train.price = train.price.str.replace('$', '').str.replace(',','').astype(float)
train = train[train.price < 10000]
train.host_response_rate = train.host_response_rate.str.replace('%','').astype(float)
train.host_acceptance_rate = train.host_acceptance_rate.str.replace('%','').astype(float)

test.host_response_rate = test.host_response_rate.str.replace('%','').astype(float)
test.host_acceptance_rate = test.host_acceptance_rate.str.replace('%','').astype(float)

# Host Total Listings Count
train['host_total_listings_count'] = train['host_total_listings_count'].apply(lambda count: 1 if count <= 7 else 0)
test['host_total_listings_count'] = test['host_total_listings_count'].apply(lambda count: 1 if count <= 7 else 0)

# Host Response Time
train['host_response_time'] = train['host_response_time'].apply(lambda time: 1 if time in ['within an hour', 'within a few hours'] else 0)
test['host_response_time'] = test['host_response_time'].apply(lambda time: 1 if time in ['within an hour', 'within a few hours'] else 0)

# Room Type
train['room_type'] = train['room_type'].apply(lambda room: 1 if room in ['Private room', 'Entire home/apt', 'Hotel room'] else 0)
test['room_type'] = test['room_type'].apply(lambda room: 1 if room in ['Private room', 'Entire home/apt', 'Hotel room'] else 0)

# Property Type
train['property_type'] = train['property_type'].apply(lambda location: 1 if location in ['Shared room in home', 'Shared room in bungalow', 'Shared room in hostel', 'Shared room in rental unit', 'Shared room in condo'] else 0)
test['property_type'] = test['property_type'].apply(lambda location: 1 if location in ['Shared room in home', 'Shared room in bungalow', 'Shared room in hostel', 'Shared room in rental unit', 'Shared room in condo'] else 0)

# Host Neighbourhood
train['host_neighbourhood'] = train['host_neighbourhood'].apply(lambda neighborhood: 1 if neighborhood in ['Lakeview', 'Mount Greenwood', 'Gold Coast', 'Chicago Loop', 'Edison Park'] else 0)
test['host_neighbourhood'] = test['host_neighbourhood'].apply(lambda neighborhood: 1 if neighborhood in ['Lakeview', 'Mount Greenwood', 'Gold Coast', 'Chicago Loop', 'Edison Park'] else 0)

# Bathrooms Text
train['bathrooms_text'] = train['bathrooms_text'].str.extract(r'(\d+)').astype(float)
test['bathrooms_text'] = test['bathrooms_text'].str.extract(r'(\d+)').astype(float)
train['bathrooms_text'] = train['bathrooms_text'].fillna(train['bathrooms_text'].median())
test['bathrooms_text'] = test['bathrooms_text'].fillna(test['bathrooms_text'].median())

# Convert float values to string
train['host_location'] = train['host_location'].astype(str)
test['host_location'] = test['host_location'].astype(str)

# City list
city_list = ['Chicago, IL', 'Lakeview', 'Mount Greenwood', 'Gold Coast', 'Chicago Loop', 'Edison Park']

# Map locations to 'West' (1), 'East' (0), and NaN (0)
train['host_location'] = train['host_location'].apply(lambda location: 1 if any(city in location for city in city_list) else (0 if location == 'nan' else 0))
test['host_location'] = test['host_location'].apply(lambda location: 1 if any(city in location for city in city_list) else (0 if location == 'nan' else 0))

# Map 't' to 1 and 'f' to 0 for binary columns
train['host_is_superhost'] = train['host_is_superhost'].map({'t': 1, 'f': 0})
test['host_is_superhost'] = test['host_is_superhost'].map({'t': 1, 'f': 0})

train['host_has_profile_pic'] = train['host_has_profile_pic'].map({'t': 1, 'f': 0})
test['host_has_profile_pic'] = test['host_has_profile_pic'].map({'t': 1, 'f': 0})

train['host_identity_verified'] = train['host_identity_verified'].map({'t': 1, 'f': 0})
test['host_identity_verified'] = test['host_identity_verified'].map({'t': 1, 'f': 0})

train['has_availability'] = train['has_availability'].map({'t': 1, 'f': 0})
test['has_availability'] = test['has_availability'].map({'t': 1, 'f': 0})

train['instant_bookable'] = train['instant_bookable'].map({'t': 1, 'f': 0})
test['instant_bookable'] = test['instant_bookable'].map({'t': 1, 'f': 0})

# Extract email or phone verification from host_verifications
train['host_verifications'] = train['host_verifications'].apply(lambda verifications: 1 if 'email' in verifications or 'phone' in verifications else 0)
test['host_verifications'] = test['host_verifications'].apply(lambda verifications: 1 if 'email' in verifications or 'phone' in verifications else 0)

# Map selected neighborhoods to 1 and others to 0
neighborhoods_to_map_to_1 = ['Lake View', 'Lincoln Park', 'Near North Side', 'West Town', 'Logan Square']
train['neighbourhood_cleansed'] = train['neighbourhood_cleansed'].apply(lambda neighborhood: 1 if neighborhood in neighborhoods_to_map_to_1 else 0)
test['neighbourhood_cleansed'] = test['neighbourhood_cleansed'].apply(lambda neighborhood: 1 if neighborhood in neighborhoods_to_map_to_1 else 0)

# Drop specified columns
columns_to_exclude = ['host_id', 'host_since', 'first_review', 'last_review']
train = train.drop(columns=columns_to_exclude, errors='ignore')
test = test.drop(columns=columns_to_exclude, errors='ignore')

train = train.fillna(train.median())
test = test.fillna(test.median())

X_train = train.drop(columns='price')
y_train = train['price']
